1.选择表和字段
```json
{ 
  "action":"select",
  "contents":{ 
    "tableId":123,//表ID
    "columns":["name","age"]//用户选择的列名
  }
}
```

2.筛选
```json
{
  "action":"where",
  "contents":[{
    "leftExpression":{
      "type":"column",//类型 可选:column-表列;value-值;multiValue-多值表达式;expression-表达式;
      "content":"age",//值 当type为column是 此值为列名;当type是value时 此值为具体值;当type是multiValue时 此值为','分隔的字符串;当type为expression时 此值为公式模板
      "valueType":"",//值类型 当type是value时要有此字段 可选:long-整型数值;string-字符串;double-浮点型数值;date-时间;null-空值;time-时间
      "params":[{}],//当type为expression时有此值 格式见4的params
      "apiParams":{
        "desc":"",//描述信息 【API使用】
        "example":"",//示例参数 【API使用】
        "require":true//是否默认 默认true【API使用】
      }
    },//当type为expression时 只有左表达式有效
    "symbol":"eq",//符号 可选:eq-等于;ne-不等于;gt-大于;gte-大于等于;lt-小于;lte-小于等于;like-字符相似(已修改为start-以...开头、end-以...结尾、contain-包含);between-介于...之间;in-在...中;exp-自定义公式;isnull-为空;notnull-不为空;
    "rightExpression":{},//右表达式 结构同左表达式
    "logic":"AND",//与前表达式的逻辑值(第一个条件不需要此表达式) 可选:AND-和;OR-或
    "nesting":[{}],//嵌套表达式 与()内的表达式嵌套 结构和本身相同
    "betweenStartExpression":{},//between开始条件 结构等同于leftExpression
    "betweenEndExpression":{}//between结束条件 结构等同于leftExpression
  }]
}
```

3.排序
```json
{
  "action":"order",
  "contents":[{
    "column":"age",//参与排序的列
    "rule":"DESC"//排序规则 可选:ASC-升序;DESC-降序
  }]
}
```

4.添加列
```json
{
  "action":"add",
  "contents":[{
    "columnAlias":"ext",//添加的列别名
    "columnName":"extColumn",//添加的列名(注意重复问题)
    "expression":"CONCAT(${1},${2})",//表达式模板
    "params":[{
      "type":"value",
      "content":"123",
      "valueType":"string"
    },{
      "type":"column",
      "content":"age"
    }
    ]//参数列表 结构同2中表达式
  }]
}
```

5.分组汇总
```json
{
  "action":"group",
  "contents":{
    "columns":["name"],//参与分组的列名
    "aggregate":[{
      "method":"MAX",//汇总方法 可选:MAX-最大值;MIN-最小值;COUNT-计数;AVG-平均值;SUM-求和;GROUP_CONCAT-字符串拼接;DISTINCT_COUNT-去重计数;
      "column":"age"//统计的列名
    }]//汇总字段
  }
}
```

6.字段设置
```json
{
   "action":"setting",
   "contents":[{
     "column":"age",//选择字段
     "columnAlias":"年龄",//列别名
     "targetFormat":"string",//目标格式 可选:string-字符;date-时间;numeric-数字
   }]//选择的字段 字段排序
}
```

7.多表合并
```json
{
  "action":"join",
  "contents":{
    "tableId":1234,//关联表ID
    "columns":[{"targetColumn":"age","columnName":"age","columnAlias":"年龄"}],//要显示关联表的字段(注意名称重复问题)
    "type":"left",//关联方法 可选:left-左合并;right-右合并;full-并集合并;inner-交集合并;union all-上下合并(只有关联的字段会出现在最终结果中);
    "onColumns":[{
      "originColumn":"old",//来源表字段
      "targetColumn":"age"//目标表字段
    }]
  }
}
```
